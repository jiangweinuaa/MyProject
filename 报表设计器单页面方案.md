# 报表设计器 - 对话式 AI 设计方案

**版本**：v3.0  
**创建时间**：2026-04-19  
**核心理念**：对话驱动 + 实时预览 + 人工干预

---

## 一、设计原则

1. **对话驱动** - 所有操作通过自然语言对话完成
2. **实时预览** - 右侧即时展示效果（SQL/数据/图表）
3. **人工干预** - AI 调整不准时，可手动编辑
4. **9 步流程** - AI 引导用户完成 9 个步骤，但用户无感知

---

### 3.2 核心服务类

```java
package com.report.service;

import org.springframework.stereotype.Service;
import java.util.Map;

/**
 * 报表设计器服务 - 统一处理 9 个步骤
 */
@Service
public class ReportDesignerService {
    
    /**
     * 执行设计器操作
     * @param sessionId 会话 ID
     * @param step 当前步骤 (1-9)
     * @param action 操作类型
     * @param params 操作参数
     * @param context 上下文（包含之前步骤的结果）
     * @return 执行结果
     */
    public Map<String, Object> execute(
        String sessionId,
        int step,
        String action,
        Map<String, Object> params,
        Map<String, Object> context
    ) {
        // 根据步骤分发处理
        return switch (step) {
            case 1 -> handleDescribe(params);           // 描述需求
            case 2 -> handleGenerateSQL(params, context); // 生成 SQL
            case 3 -> handlePreviewData(params, context); // 预览数据
            case 4 -> handleSelectChart(params, context); // 选择图表
            case 5 -> handleAdjustStyle(params, context); // 调整样式
            case 6 -> handleAddFilters(params, context);  // 添加筛选
            case 7 -> handleSetLayout(params, context);   // 设置布局
            case 8 -> handleSaveReport(params, context);  // 命名保存
            case 9 -> handleSelectMenu(params, context);  // 选择目录
            default -> error("未知步骤：" + step);
        };
    }
    
    // ========== 各步骤处理方法 ==========
    
    private Map<String, Object> handleDescribe(Map<String, Object> params) {
        // 步骤 1：理解用户需求，生成查询意图
        String question = (String) params.get("question");
        
        // 调用 AI 理解意图
        Map<String, Object> understood = aiService.understandQuestion(question);
        
        return Map.of(
            "success", true,
            "nextStep", 2,
            "data", understood,
            "message", "已理解您的需求"
        );
    }
    
    private Map<String, Object> handleGenerateSQL(
        Map<String, Object> params,
        Map<String, Object> context
    ) {
        // 步骤 2：生成 SQL
        String question = (String) context.get("question");
        String sql = aiService.generateSQL(question);
        
        // SQL 验证
        boolean valid = sqlValidator.validate(sql);
        
        return Map.of(
            "success", valid,
            "nextStep", valid ? 3 : 2,
            "data", Map.of("sql", sql),
            "message", valid ? "SQL 生成成功" : "SQL 生成失败，请重试"
        );
    }
    
    private Map<String, Object> handlePreviewData(
        Map<String, Object> params,
        Map<String, Object> context
    ) {
        // 步骤 3：执行 SQL 预览数据
        String sql = (String) context.get("sql");
        Map<String, Object> data = dataService.execute(sql, 10); // 限制 10 条
        
        return Map.of(
            "success", true,
            "nextStep", 4,
            "data", data,
            "message", "数据预览完成"
        );
    }
    
    private Map<String, Object> handleSelectChart(
        Map<String, Object> params,
        Map<String, Object> context
    ) {
        // 步骤 4：选择/生成图表
        String chartType = (String) params.get("chartType");
        Map<String, Object> data = (Map<String, Object>) context.get("data");
        
        Map<String, Object> chartConfig = aiService.generateChartConfig(
            chartType, data
        );
        
        return Map.of(
            "success", true,
            "nextStep", 5,
            "data", Map.of("chartConfig", chartConfig),
            "message", "图表生成成功"
        );
    }
    
    private Map<String, Object> handleAdjustStyle(
        Map<String, Object> params,
        Map<String, Object> context
    ) {
        // 步骤 5：调整样式
        Map<String, Object> currentConfig = 
            (Map<String, Object>) context.get("chartConfig");
        
        // 应用样式调整
        Map<String, Object> newConfig = chartStyler.apply(
            currentConfig, params
        );
        
        return Map.of(
            "success", true,
            "nextStep", 6,
            "data", Map.of("chartConfig", newConfig),
            "message", "样式调整完成"
        );
    }
    
    private Map<String, Object> handleAddFilters(
        Map<String, Object> params,
        Map<String, Object> context
    ) {
        // 步骤 6：添加筛选条件
        String sql = (String) context.get("sql");
        List<Map<String, Object>> filters = 
            (List<Map<String, Object>>) params.get("filters");
        
        // 修改 SQL 添加变量
        String newSql = sqlModifier.addVariables(sql, filters);
        
        return Map.of(
            "success", true,
            "nextStep", 7,
            "data", Map.of("sql", newSql, "variables", filters),
            "message", "筛选条件已添加"
        );
    }
    
    private Map<String, Object> handleSetLayout(
        Map<String, Object> params,
        Map<String, Object> context
    ) {
        // 步骤 7：设置布局
        Map<String, Object> layout = layoutBuilder.build(params);
        
        return Map.of(
            "success", true,
            "nextStep", 8,
            "data", Map.of("layout", layout),
            "message", "布局设置完成"
        );
    }
    
    private Map<String, Object> handleSaveReport(
        Map<String, Object> params,
        Map<String, Object> context
    ) {
        // 步骤 8：保存报表
        String reportName = (String) params.get("reportName");
        String description = (String) params.get("description");
        
        // 组装完整报表配置
        Map<String, Object> report = assembleReport(params, context);
        
        // 保存到数据库
        String reportId = reportRepository.save(report);
        
        return Map.of(
            "success", true,
            "nextStep", 9,
            "data", Map.of("reportId", reportId),
            "message", "报表保存成功"
        );
    }
    
    private Map<String, Object> handleSelectMenu(
        Map<String, Object> params,
        Map<String, Object> context
    ) {
        // 步骤 9：选择目录
        String reportId = (String) context.get("reportId");
        String menuId = (String) params.get("menuId");
        
        // 关联报表到菜单
        menuRepository.addReport(menuId, reportId);
        
        return Map.of(
            "success", true,
            "nextStep", 0, // 0 表示完成
            "data", Map.of("reportId", reportId, "menuId", menuId),
            "message", "报表创建完成！"
        );
    }
}
```

### 3.3 会话管理

```java
package com.report.service;

import org.springframework.stereotype.Service;
import java.util.concurrent.ConcurrentHashMap;

/**
 * 设计器会话管理
 */
@Service
public class DesignerSessionManager {
    
    // 内存存储会话（生产环境建议用 Redis）
    private final ConcurrentHashMap<String, DesignerSession> sessions = 
        new ConcurrentHashMap<>();
    
    /**
     * 创建新会话
     */
    public DesignerSession createSession() {
        String sessionId = "DESIGN_" + System.currentTimeMillis();
        DesignerSession session = new DesignerSession(sessionId);
        sessions.put(sessionId, session);
        return session;
    }
    
    /**
     * 获取会话
     */
    public DesignerSession getSession(String sessionId) {
        return sessions.get(sessionId);
    }
    
    /**
     * 更新会话上下文
     */
    public void updateContext(String sessionId, Map<String, Object> context) {
        DesignerSession session = sessions.get(sessionId);
        if (session != null) {
            session.getContext().putAll(context);
        }
    }
    
    /**
     * 会话过期清理（定时任务）
     */
    @Scheduled(fixedRate = 3600000) // 每小时
    public void cleanupExpiredSessions() {
        long now = System.currentTimeMillis();
        sessions.entrySet().removeIf(entry -> 
            now - entry.getValue().getLastAccessTime() > 86400000 // 24 小时
        );
    }
}

/**
 * 设计器会话对象
 */
@Data
public class DesignerSession {
    private String sessionId;
    private int currentStep = 1;
    private Map<String, Object> context = new HashMap<>();
    private long lastAccessTime = System.currentTimeMillis();
    
    public DesignerSession(String sessionId) {
        this.sessionId = sessionId;
    }
}
```

### 4.2 状态管理

```javascript
// 使用 Pinia 管理设计器状态
import { defineStore } from 'pinia'

export const useDesignerStore = defineStore('designer', {
  state: () => ({
    sessionId: null,
    currentStep: 1,
    completedSteps: [],
    context: {
      question: '',
      sql: '',
      data: null,
      chartType: '',
      chartConfig: null,
      filters: [],
      layout: null,
      reportName: '',
      menuId: ''
    },
    isSaving: false,
    isProcessing: false
  }),
  
  actions: {
    async executeStep(action, params) {
      this.isProcessing = true
      
      try {
        const response = await fetch('/api/designer/execute', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            sessionId: this.sessionId,
            step: this.currentStep,
            action,
            params,
            context: this.context
          })
        })
        
        const result = await response.json()
        
        if (result.success) {
          // 更新上下文
          Object.assign(this.context, result.data)
          
          // 进入下一步
          if (result.nextStep > 0) {
            this.completedSteps.push(this.currentStep)
            this.currentStep = result.nextStep
          }
        }
        
        return result
      } finally {
        this.isProcessing = false
      }
    },
    
    goToStep(step) {
      if (step < this.currentStep || this.completedSteps.includes(step - 1)) {
        this.currentStep = step
      }
    }
  }
})
```
